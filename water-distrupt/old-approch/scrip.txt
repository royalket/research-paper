def _cmc_to_year_month(cmc: pd.Series) -> pd.DataFrame:
    """Convert DHS Century Month Code (hv008) to year and month."""
    year = ((cmc.astype("float") - 1) // 12 + 1900).astype("Int64")
    month = (cmc.astype("float") - (year - 1900) * 12 + 1).astype("Int64")
    return pd.DataFrame({"year": year, "month": month})

def _assign_india_season(month: pd.Series) -> pd.Series:
    """Map month number (1-12) to broad Indian seasons."""
    conditions = [
        month.isin([12, 1, 2]),
        month.isin([3, 4, 5]),
        month.isin([6, 7, 8, 9]),
        month.isin([10, 11]),
    ]
    choices = ["winter", "summer", "monsoon", "post_monsoon"]
    return pd.Series(np.select(conditions, choices, default=pd.NA), index=month.index).astype("string")

def _binary_from_codes(s: pd.Series, yes_codes=(1,), no_codes=(0,), missing_codes=(8, 9, 99)) -> pd.Series:
    """Map code lists to 0.0/1.0 float with NaN for missing/other."""
    if s is None or s.empty:
        return pd.Series(np.nan, index=pd.RangeIndex(len(s) if s is not None else 0), dtype="float")
    
    out = pd.Series(np.nan, index=s.index, dtype="float")
    s_numeric = pd.to_numeric(s, errors='coerce')
    out = out.mask(s_numeric.isin(no_codes), 0.0) # Use 0.0 for False
    out = out.mask(s_numeric.isin(yes_codes), 1.0) # Use 1.0 for True
    out = out.mask(s_numeric.isna(), np.nan)
    out = out.mask(s_numeric.isin(missing_codes), np.nan)
    return out

def _on_premises_from_hv235(hv235: pd.Series) -> pd.Series:
    """DHS hv235: Location of source for water."""
    if hv235 is None:
        return pd.Series(pd.NA, index=pd.RangeIndex(0), dtype="boolean")
    return _binary_from_codes(hv235, yes_codes=(11, 12), no_codes=(13,))

def _minutes_to_source(hv204: pd.Series, on_premises: Optional[pd.Series] = None) -> pd.Series:
    """Minutes to water source; if on premises, set to 0."""
    mins = pd.to_numeric(hv204, errors="coerce")
    if on_premises is not None:
        mins = mins.mask(on_premises == True, 0)
    mins = mins.replace([997, 998, 999], np.nan)
    return mins

def _safe_treated_water(hv237: pd.Series) -> pd.Series:
    """DHS hv237: Anything done to water to make it safer to drink."""
    if hv237 is None:
        return pd.Series(pd.NA, index=pd.RangeIndex(0), dtype="boolean")
    return _binary_from_codes(hv237, yes_codes=(1,), no_codes=(0,))

def _extract_head_education(df: pd.DataFrame) -> pd.Series:
    """Extracts the education level of the household head."""
    head_edu = pd.Series(np.nan, index=df.index, dtype='float')
    for i in range(1, 21):
        rel_col = f'hv101_{i:02d}'
        edu_col = f'hv106_{i:02d}'
        if rel_col in df.columns and edu_col in df.columns:
            is_head = (df[rel_col] == 1)
            head_edu = head_edu.fillna(df.loc[is_head, edu_col])

    edu_map = {0: 'No education', 1: 'Primary', 2: 'Secondary', 3: 'Higher', 4: 'Higher', 8: 'Don\'t Know', 9: 'Missing'}
    return head_edu.map(edu_map).fillna('Unknown')

def safe_weighted_average(group_df, value_col, weight_col):
    """Calculates weighted average, handling cases where weights sum to zero or no valid data."""
    valid_subset = group_df[value_col].notna() & group_df[weight_col].notna()
    if not valid_subset.any():
        return np.nan
    
    subset_values = group_df.loc[valid_subset, value_col]
    subset_weights = group_df.loc[valid_subset, weight_col]

    if subset_weights.sum() == 0:
        return np.nan
    
    try:
        return np.average(subset_values, weights=subset_weights)
    except Exception:
        return np.nan

class EnhancedDataLoader:
    """Enhanced data loader with capability to read DTA files and extract metadata."""
    
    def __init__(self, config: Config):
        self.config = config
        self.metadata: Optional[Any] = None

    def load(self, filepath: str) -> pd.DataFrame:
        """Load data with specified columns and store metadata for value labels."""
        print(f"Loading NFHS-5 data for comprehensive water disruption analysis from {filepath}...")
        
        df = pd.DataFrame()
        meta = None
        try:
            df, meta = pyreadstat.read_dta(filepath, usecols=REQUIRED_COLS)
            print(f"Successfully loaded {len(REQUIRED_COLS)} specified columns.")
        except Exception as e:
            print(f"Error loading specified columns: {e}. Attempting to load all available columns...")
            try:
                df, meta = pyreadstat.read_dta(filepath)
                available_cols = [col for col in REQUIRED_COLS if col in df.columns]
                missing_cols = [col for col in REQUIRED_COLS if col not in df.columns]
                df = df[available_cols]
                print(f"Loaded {len(available_cols)} out of {len(REQUIRED_COLS)} requested columns.")
                if missing_cols:
                    print(f"Missing columns: {missing_cols}")
            except Exception as e_all:
                print(f"Failed to load any data: {e_all}")
                return pd.DataFrame()
        self.metadata = meta
        df.attrs['metadata'] = meta
        
        print(f"Loaded {len(df):,} households with {len(df.columns)} variables.")
        
        water_vars_to_check = {'sh37b': 'PRIMARY (sh37b)', 'hv201a': 'ADDITIONAL (hv201a)'}
        for var_name, description in water_vars_to_check.items():
            if var_name in df.columns:
                print(f"\nDisruption variable {description} found:")
                value_counts = df[var_name].value_counts(dropna=False).sort_index()
                total = len(df)
                var_labels = meta.variable_value_labels.get(var_name, {})
                for val, count in value_counts.items():
                    pct = count/total*100
                    label = var_labels.get(val, f"Code {val}")
                    print(f"  {label}: {count:,} ({pct:.1f}%)")
            else:
                print(f"\nDisruption variable {description} not found in data.")
                
        return df

                try:
                    state_code_int = int(state_code_val)
                    for region_name, state_codes in REGIONS.items():
                        if state_code_int in state_codes: return region_name
                    return 'Other'
                except ValueError: return 'Unknown'
            self.df['region'] = self.df['state_code'].apply(get_region)

            self.df['state_name'] = pd.NA
            self.df['state_code'] = pd.NA

                var_data['categories'][cat_label] = {

                'missing_weighted_n': float(missing_w_n),
                'missing_weighted_percent': (missing_w_n / total_weighted) * 100 if total_weighted > 0 else 0
            }
            self.sample_sizes[var] = var_data

class BivariateAnalyzer:
    """Comprehensive bivariate statistical analysis (unweighted)."""
    
    def __init__(self, df: pd.DataFrame, config: Config):
        self.df = df
        self.config = config

    def run_comprehensive_bivariate_analysis(self) -> Dict:
        """Run all bivariate analyses."""
        print("\nRunning comprehensive bivariate analysis (unweighted)...")
        
        results = {}
        results['chi_square_tests'] = self._chi_square_tests()
        results['t_tests'] = self._t_tests()
        results['anova_tests'] = self._anova_tests()
        results['correlation_analysis'] = self._correlation_analysis()
        results['rank_correlations'] = self._rank_correlations()
        results['contingency_tables'] = self._contingency_tables()
        
        print("Bivariate analysis complete.")
        return results

    def _chi_square_tests(self) -> Dict:
        """Comprehensive chi-square tests for independence with Cramer's V."""
        cat_vars = [
            'residence_type', 'wealth_quintile', 'region', 'season', 'water_source_category',
            'caste_category', 'religion', 'hh_head_education_level', 'toilet_type',
            'time_to_water_category', 'water_fetcher', 'water_source_location',
            'place_type', 'has_improved_source', 'women_fetch_water', 'children_fetch_water',
            'has_electricity', 'has_improved_sanitation', 'practices_open_defecation',
            'has_bank_account', 'has_separate_kitchen', 'has_vehicle', 'has_finished_floor',
            'has_finished_wall', 'has_finished_roof',
            'treats_water_safe'
        ]
        
        results = {}
        target_var = 'water_disrupted'
        if target_var not in self.df.columns:
            return {'error': f'Target variable "{target_var}" not found.'}

        for var in cat_vars:
            if var not in self.df.columns:
                continue
            
            temp_df = self.df[[var, target_var]].dropna()
            
            if len(temp_df) < self.config.min_sample:
                continue
            crosstab = pd.crosstab(temp_df[var], temp_df[target_var])
            
            if crosstab.empty or min(crosstab.shape) < 2:
                continue
            try:
                chi2, p_val, dof, expected = chi2_contingency(crosstab)
                
                n = crosstab.sum().sum()
                min_dim = min(crosstab.shape)
                if n > 0 and min_dim > 1:
                    cramer_v = np.sqrt(chi2 / (n * (min_dim - 1)))
                else:
                    cramer_v = np.nan
                
                results[var] = {
                    'chi2_statistic': chi2,
                    'p_value': p_val,
                    'degrees_freedom': dof,
                    'cramers_v': cramer_v,
                    'sample_size': n,
                    'significant': p_val < self.config.alpha
                }
            except Exception as e:
                results[var] = {'error': str(e), 'crosstab_shape': crosstab.shape, 'sample_size': len(temp_df)}
        
        return results

    def _t_tests(self) -> Dict:
        """Independent samples t-tests comparing continuous variables across disruption groups."""
        continuous_vars = [
            'wealth_score_continuous', 'hh_size', 'children_under5_count',
            'infrastructure_asset_score', 'housing_quality_score',
            'water_disruptive_score', 'water_accessibility_score',
            'vulnerability_index', 'coping_capabilities_index', 'time_to_water_minutes'
        ]
        
        results = {}
        target_var = 'water_disrupted'
        if target_var not in self.df.columns:
            return {'error': f'Target variable "{target_var}" not found.'}

        for var in continuous_vars:
            if var not in self.df.columns:
                continue
            
            temp_df = self.df[[var, target_var]].dropna()
            
            group0 = temp_df[temp_df[target_var] == False][var].dropna()
            group1 = temp_df[temp_df[target_var] == True][var].dropna()
            
            if len(group0) >= self.config.min_sample and len(group1) >= self.config.min_sample:
                try:
                    t_stat, p_val = ttest_ind(group0, group1, equal_var=False)
                    
                    pooled_std = np.sqrt(((len(group0) - 1) * group0.var() + (len(group1) - 1) * group1.var()) / \
                                          (len(group0) + len(group1) - 2)) if (len(group0) + len(group1) - 2) > 0 else 0
                    cohens_d = (group1.mean() - group0.mean()) / pooled_std if pooled_std > 0 else 0
                    
                    results[var] = {
                        'group0_mean': group0.mean(),
                        'group1_mean': group1.mean(),
                        'group0_std': group0.std(),
                        'group1_std': group1.std(),
                        'group0_n': len(group0),
                        'group1_n': len(group1),
                        't_statistic': t_stat,
                        'p_value': p_val,
                        'cohens_d': cohens_d,
                        'significant': p_val < self.config.alpha
                    }
                except Exception as e:
                    results[var] = {'error': str(e), 'sample_sizes': (len(group0), len(group1))}
        
        return results

    def _anova_tests(self) -> Dict:
        """One-way ANOVA tests comparing continuous variables across multiple categorical groups."""
        continuous_vars = [
            'wealth_score_continuous', 'hh_size', 'children_under5_count',
            'infrastructure_asset_score', 'housing_quality_score',
            'water_disruptive_score', 'water_accessibility_score',
            'vulnerability_index', 'coping_capabilities_index', 'time_to_water_minutes'
        ]
        categorical_vars = [
            'residence_type', 'wealth_quintile', 'region', 'season', 'water_source_category',
            'caste_category', 'religion', 'hh_head_education_level', 'toilet_type',
            'time_to_water_category', 'water_fetcher', 'water_source_location',
            'place_type'
        ]
        
        results = {}
        
        for cat_var in categorical_vars:
            if cat_var not in self.df.columns:
                continue
            for cont_var in continuous_vars:
                if cont_var not in self.df.columns:
                    continue
                
                temp_df = self.df[[cont_var, cat_var]].dropna()
                if len(temp_df) < self.config.min_sample:
                    continue
                groups = []
                group_names = []
                for category in temp_df[cat_var].unique():
                    group_data = temp_df[temp_df[cat_var] == category][cont_var].dropna()
                    if len(group_data) >= self.config.min_sample:
                        groups.append(group_data)
                        group_names.append(category)
                
                if len(groups) >= 2:
                    try:
                        f_stat, f_p_val = f_oneway(*groups)
                        
                        h_stat, h_p_val = kruskal(*groups)
                        
                        all_data = np.concatenate(groups)
                        ss_total = np.sum((all_data - np.mean(all_data)) ** 2)
                        ss_between = sum([len(g) * (np.mean(g) - np.mean(all_data)) ** 2 for g in groups])
                        eta_squared = ss_between / ss_total if ss_total > 0 else 0
                        
                        results[f"{cont_var}_by_{cat_var}"] = {
                            'f_statistic': f_stat,
                            'f_p_value': f_p_val,
                            'kruskal_h': h_stat,
                            'kruskal_p': h_p_val,
                            'eta_squared': eta_squared,
                            'groups_analyzed': group_names,
                            'group_means': {name: group.mean() for name, group in zip(group_names, groups)},
                            'significant_anova': f_p_val < self.config.alpha,
                            'significant_kruskal': h_p_val < self.config.alpha
                        }
                    except Exception as e:
                        results[f"{cont_var}_by_{cat_var}"] = {'error': str(e), 'groups_info': {name: len(g) for name, g in zip(group_names, groups)}}
        
        return results

    def _correlation_analysis(self) -> Dict:
        """Comprehensive correlation analysis (Pearson, Spearman, Kendall) for numerical pairs."""
        numerical_vars = [
            'wealth_score_continuous', 'hh_size', 'children_under5_count',
            'infrastructure_asset_score', 'housing_quality_score', 'time_to_water_minutes',
            'water_disruptive_score', 'water_accessibility_score',
            'vulnerability_index', 'coping_capabilities_index',
            'water_disrupted'
        ]
        
        results = {}
        available_numeric_vars = [v for v in numerical_vars if v in self.df.columns]
        
        if len(available_numeric_vars) < 2:
            return {'error': 'Insufficient numerical variables for correlation analysis.'}
        
        corr_matrix_pearson = pd.DataFrame(index=available_numeric_vars, columns=available_numeric_vars, dtype=float)
        p_value_matrix_pearson = pd.DataFrame(index=available_numeric_vars, columns=available_numeric_vars, dtype=float)
        corr_matrix_spearman = pd.DataFrame(index=available_numeric_vars, columns=available_numeric_vars, dtype=float)
        p_value_matrix_spearman = pd.DataFrame(index=available_numeric_vars, columns=available_numeric_vars, dtype=float)
        corr_matrix_kendall = pd.DataFrame(index=available_numeric_vars, columns=available_numeric_vars, dtype=float)
        p_value_matrix_kendall = pd.DataFrame(index=available_numeric_vars, columns=available_numeric_vars, dtype=float)

        for i, var1 in enumerate(available_numeric_vars):
            for j, var2 in enumerate(available_numeric_vars):
                if i == j:
                    corr_matrix_pearson.loc[var1, var2] = 1.0
                    p_value_matrix_pearson.loc[var1, var2] = 0.0
                    corr_matrix_spearman.loc[var1, var2] = 1.0
                    p_value_matrix_spearman.loc[var1, var2] = 0.0
                    corr_matrix_kendall.loc[var1, var2] = 1.0
                    p_value_matrix_kendall.loc[var1, var2] = 0.0
                elif i < j:
                    temp_df = self.df[[var1, var2]].dropna()
                    if len(temp_df) >= self.config.min_sample:
                        try:
                            r_p, p_p = pearsonr(temp_df[var1], temp_df[var2])
                            corr_matrix_pearson.loc[var1, var2] = r_p
                            corr_matrix_pearson.loc[var2, var1] = r_p
                            p_value_matrix_pearson.loc[var1, var2] = p_p
                            p_value_matrix_pearson.loc[var2, var1] = p_p
                            r_s, p_s = spearmanr(temp_df[var1], temp_df[var2])
                            corr_matrix_spearman.loc[var1, var2] = r_s
                            corr_matrix_spearman.loc[var2, var1] = r_s
                            p_value_matrix_spearman.loc[var1, var2] = p_s
                            p_value_matrix_spearman.loc[var2, var1] = p_s
                            r_k, p_k = kendalltau(temp_df[var1], temp_df[var2])
                            corr_matrix_kendall.loc[var1, var2] = r_k
                            corr_matrix_kendall.loc[var2, var1] = r_k
                            p_value_matrix_kendall.loc[var1, var2] = p_k
                            p_value_matrix_kendall.loc[var2, var1] = p_k
                        except Exception as e:
                            corr_matrix_pearson.loc[var1, var2] = np.nan
                            corr_matrix_pearson.loc[var2, var1] = np.nan
                            p_value_matrix_pearson.loc[var1, var2] = np.nan
                            p_value_matrix_pearson.loc[var2, var1] = np.nan
                    else:
                        corr_matrix_pearson.loc[var1, var2] = np.nan
                        corr_matrix_pearson.loc[var2, var1] = np.nan
                        p_value_matrix_pearson.loc[var1, var2] = np.nan
                        p_value_matrix_pearson.loc[var2, var1] = np.nan
        
        results = {
            'pearson_correlation_matrix': corr_matrix_pearson,
            'pearson_p_value_matrix': p_value_matrix_pearson,
            'spearman_correlation_matrix': corr_matrix_spearman,
            'spearman_p_value_matrix': p_value_matrix_spearman,
            'kendall_correlation_matrix': corr_matrix_kendall,
            'kendall_p_value_matrix': p_value_matrix_kendall,
            'sample_size_for_pairwise_correlations': len(self.df)
        }
        return results

    def _contingency_tables(self) -> Dict:
        """Generate detailed contingency tables for key categorical variables vs. water_disrupted."""
        results = {}
        target_var = 'water_disrupted'
        if target_var not in self.df.columns:
            return {'error': f'Target variable "{target_var}" not found.'}

        cat_vars_for_crosstab = [
            'residence_type', 'wealth_quintile', 'region', 'season', 'water_source_category',
            'caste_category', 'religion', 'hh_head_education_level', 'toilet_type',
            'time_to_water_category', 'water_fetcher', 'water_source_location',
            'place_type'
        ]
        
        for var in cat_vars_for_crosstab:
            if var not in self.df.columns:
                continue
            
            temp_df = self.df[[var, target_var]].dropna()
            
            if len(temp_df) < self.config.min_sample:
                continue
            crosstab_counts = pd.crosstab(temp_df[var], temp_df[target_var], margins=True, dropna=False)
            
            crosstab_row_pct = pd.crosstab(temp_df[var], temp_df[target_var], normalize='index', margins=True, dropna=False) * 100
            
            crosstab_col_pct = pd.crosstab(temp_df[var], temp_df[target_var], normalize='columns', margins=True, dropna=False) * 100
            
            crosstab_total_pct = pd.crosstab(temp_df[var], temp_df[target_var], normalize='all', margins=True, dropna=False) * 100
            
            results[var] = {
                'counts': crosstab_counts,
                'row_percentages': crosstab_row_pct,
                'column_percentages': crosstab_col_pct,
                'total_percentages': crosstab_total_pct,
                'n_observations': len(temp_df)
            }
        
        return results

    def _logistic_regression_analysis(self) -> Dict:
        """Comprehensive logistic regression predicting water_disrupted."""
        print("  - Running logistic regression...")
        try:
            predictors = [
                'water_accessibility_score', 'vulnerability_index', 'coping_capabilities_index',
                'wealth_score_continuous', 'hh_size', 'children_under5_count',
                'has_improved_source', 'water_on_premises', 'treats_water_safe',
                'female_headed_household', 'marginalized_caste', 'hh_head_no_education'
            ]
            
            categorical_predictors = [
                'residence_type', 'season', 'water_source_category', 'toilet_type', 'region'
            ]
            
            model_df = self.df.copy()
            
            for p in predictors:
                if p in model_df.columns:
                    model_df[p] = pd.to_numeric(model_df[p], errors='coerce').replace([np.inf, -np.inf], np.nan)

            formula_parts = []
            for p in predictors:
                if p in model_df.columns and model_df[p].notna().any():
                    formula_parts.append(p)
            for cp in categorical_predictors:
                if cp in model_df.columns and model_df[cp].nunique() > 1 and model_df[cp].notna().any():
                    formula_parts.append(f'C({cp})')
            
            target_var = 'water_disrupted'
            if target_var not in model_df.columns or not model_df[target_var].notna().any():
                return {'error': f'Target variable "{target_var}" is missing or all NA.'}
            
            model_df[target_var] = model_df[target_var].astype(float)
            
            if not formula_parts:
                return {'error': 'No valid predictors found for logistic regression.'}
            
            full_formula = f'{target_var} ~ ' + ' + '.join(formula_parts)
            
            cols_for_dropna = [target_var, 'original_weight', 'hv021'] + [p for p in predictors if p in model_df.columns] + [cp for cp in categorical_predictors if cp in model_df.columns]
            model_df_cleaned = model_df[cols_for_dropna].dropna()
            
            if len(model_df_cleaned) < self.config.min_sample * 5:
                return {'error': f'Insufficient data for logistic regression after NA removal ({len(model_df_cleaned)} rows).'}
            
            weights = model_df_cleaned['original_weight']
            
            model = smf.glm(
                formula=full_formula,
                data=model_df_cleaned,
                family=sm.families.Binomial(),
                freq_weights=weights
            )
            
            results_sm = model.fit(cov_type='cluster', cov_kwds={'groups': model_df_cleaned['hv021']})
            
            coef_df = pd.DataFrame({
                'Variable': results_sm.params.index,
                'Coefficient': results_sm.params.values,
                'Std_Error': results_sm.bse.values,
                'z_value': results_sm.tvalues.values,
                'p_value': results_sm.pvalues.values,
                'OR': np.exp(results_sm.params.values),
                'OR_CI_lower': np.exp(results_sm.params.values - 1.96*results_sm.bse.values),
                'OR_CI_upper': np.exp(results_sm.params.values + 1.96*results_sm.bse.values)
            })
            
            coef_df['Significance'] = coef_df['p_value'].apply(
                lambda p: '***' if p < 0.001 else '**' if p < 0.01 else '*' if p < 0.05 else ''
            )
            
            null_model = smf.glm(f'{target_var} ~ 1', data=model_df_cleaned, family=sm.families.Binomial(), freq_weights=weights)
            null_results = null_model.fit(disp=False)
            mcfadden_r2 = 1 - (results_sm.llf / null_results.llf)
            
            return {
                'coefficients_summary': coef_df,
                'model_fit_statistics': {
                    'aic': results_sm.aic,
                    'bic': results_sm.bic,
                    'log_likelihood': results_sm.llf,
                    'mcfadden_r2': mcfadden_r2,
                    'n_observations': len(model_df_cleaned),
                    'formula': full_formula
                },
                'model_summary_text': results_sm.summary().as_text()
            }
            
        except Exception as e:
            return {
                'error': f'Logistic regression failed: {str(e)}',
                'model_fit_statistics': {
                    'aic': np.nan, 'bic': np.nan, 'log_likelihood': np.nan, 'mcfadden_r2': np.nan,
                    'n_observations': len(model_df_cleaned) if 'model_df_cleaned' in locals() else 0,
                    'formula': full_formula if 'full_formula' in locals() else 'N/A'
                },
                'model_summary_text': f"Model execution failed: {str(e)}"
            }

    def _run_single_manova(self, indep_var: str, dependent_vars: List[str]) -> Dict:
        """Helper to run a single MANOVA test."""
        if indep_var not in self.df.columns or self.df[indep_var].nunique(dropna=True) < 2:
            return {'error': f'Independent variable "{indep_var}" not suitable for MANOVA (missing or <2 unique values).'}
        
        available_deps = [v for v in dependent_vars if v in self.df.columns and self.df[v].notna().any()]
        
        if len(available_deps) < 2:
            return {'error': f'Insufficient dependent variables ({len(available_deps)}) for MANOVA with {indep_var}.'}
        
        manova_data = self.df[available_deps + [indep_var, 'original_weight', 'hv021']].dropna() # Ensure weights and PSU also included
        
        if len(manova_data) < self.config.min_sample * 2:
            return {'error': f'Insufficient observations for MANOVA with {indep_var} ({len(manova_data)} rows).'}
        
        # CRITICAL: Ensure the independent variable is 'object' type for statsmodels formula API
        manova_data[indep_var] = manova_data[indep_var].astype('object')
        
        dep_formula_str = ' + '.join(available_deps)
        full_formula = f'{dep_formula_str} ~ C({indep_var})'
        
        try:
            manova = MANOVA.from_formula(full_formula, data=manova_data)
            manova_results_obj = manova.mv_test()

            parsed_tables = []
            
            def parse_summary_table(table_str, table_name):
                lines = table_str.split('\n')
                header_lines = []
                data_lines = []
                in_header = False
                in_data = False

                for line in lines:
                    if '---' in line:
                        if not in_header:
                            in_header = True
                        elif in_header and not in_data:
                            in_header = False
                            in_data = True
                            continue
                        elif in_data:
                            break
                        continue

                    if in_header:
                        header_lines.append(line.strip())
                    elif in_data:
                        data_lines.append(line.strip())
                
                if not header_lines or not data_lines:
                    return None

                header_str = " ".join(header_lines)
                header = [h.strip() for h in re.split(r'\s{2,}', header_str) if h.strip()]

                rows = []
                for line in data_lines:
                    row_data = [d.strip() for d in re.split(r'\s{2,}', line) if d.strip()]
                    if row_data:
                        rows.append(row_data)
                
                if not rows: return None

                max_cols = max(len(row) for row in rows)
                if len(header) < max_cols:
                    header.extend([f'Col_{i}' for i in range(len(header), max_cols)])
                elif len(header) > max_cols:
                    header = header[:max_cols]

                try:
                    df = pd.DataFrame(rows, columns=header)
                    return {'name': table_name, 'table': df.to_dict('records')}
                except ValueError as ve:
                    print(f"Warning: Column mismatch in parsing MANOVA table {table_name}: {ve}")
                    print(f"Header: {header}\nFirst few rows: {rows[:3]}")
                    return None

            if hasattr(manova_results_obj, 'mv_test_results') and 'Intercept' in manova_results_obj.mv_test_results:
                table = parse_summary_table(str(manova_results_obj.mv_test_results['Intercept']), 'Intercept')
                if table: parsed_tables.append(table)

            if hasattr(manova_results_obj, 'mv_test_results') and indep_var in manova_results_obj.mv_test_results:
                table = parse_summary_table(str(manova_results_obj.mv_test_results[indep_var]), indep_var)
                if table: parsed_tables.append(table)
            
            return {
                'formula': full_formula,
                'n_observations': len(manova_data),
                'dependent_variables': available_deps,
                'independent_variable': indep_var,
                'manova_summary_text': str(manova_results_obj) 
            }
        except Exception as e:
            return {'error': f'MANOVA for {indep_var} failed: {str(e)}'}

            for future in as_completed(futures):
                name = futures[future]
                try:
                    result = future.result()
                    results[name] = result
                    print(f"  - Completed {name}")
                except Exception as e:
                    results[name] = {'error': f'Failed {name}: {str(e)}'}
                    print(f"  - Failed {name}: {str(e)}")
        
        print("  - Running multiple comparisons correction (sequential due to dependency)...")
        results['multiple_comparisons_correction'] = self._multiple_comparisons_correction()
        print("  - Completed multiple comparisons correction.")
        results['spatial_autocorrelation_note'] = "Spatial autocorrelation requires specific spatial weights matrices (e.g., contiguity, distance) which are not pre-computed here. A simplified check could be implemented if cluster/PSU coordinates were available."
        
        print("Advanced statistical tests complete.")
        return results

    def _normality_tests(self) -> Dict:
        """Test normality of continuous variables using Shapiro-Wilk, D'Agostino, and K-S."""
        print("  - Running normality tests...")

        continuous_vars = [

        return results

    def _run_single_normality_test(self, var: str) -> Dict:
        """Helper to run normality tests for a single variable."""
        from scipy.stats import shapiro, normaltest, kstest
        
        if var not in self.df.columns:
            return {'error': f'Variable "{var}" not found.'}
        
        data = self.df[var].dropna()
        
        if len(data) < self.config.min_sample:
            return {'error': f'Not enough samples ({len(data)}) for meaningful test.'}
        
        shapiro_stat, shapiro_p = np.nan, np.nan
        if len(data) <= 5000 and len(data) >= 3:
            try:
                shapiro_stat, shapiro_p = shapiro(data)
            except Exception:
                shapiro_stat, shapiro_p = np.nan, np.nan
        
        dagostino_stat, dagostino_p = np.nan, np.nan
        if len(data) >= 8:
            try:
                dagostino_stat, dagostino_p = normaltest(data)
            except Exception:
                dagostino_stat, dagostino_p = np.nan, np.nan

        mean_val, std_val = data.mean(), data.std()
        ks_stat, ks_p = np.nan, np.nan
        if std_val > 0 and len(data) >= 1:
            try:
                ks_stat, ks_p = kstest(data, 'norm', args=(mean_val, std_val))
            except Exception:
                ks_stat, ks_p = np.nan, np.nan
        
        return {
            'sample_size': len(data),
            'mean': data.mean(),
            'std_dev': data.std(),
            'skewness': stats.skew(data),
            'kurtosis': stats.kurtosis(data),
            'shapiro_statistic': shapiro_stat,
            'shapiro_p_value': shapiro_p,
            'dagostino_statistic': dagostino_stat,
            'dagostino_p_value': dagostino_p,
            'ks_statistic': ks_stat,
            'ks_p_value': ks_p,
            'is_normal_shapiro': shapiro_p >= self.config.alpha if not np.isnan(shapiro_p) else None,
            'is_normal_dagostino': dagostino_p >= self.config.alpha if not np.isnan(dagostino_p) else None,
            'is_normal_ks': ks_p >= self.config.alpha if not np.isnan(ks_p) else None
        }

    def _homogeneity_tests(self) -> Dict:
        """Test homogeneity of variances using Levene's and Bartlett's tests."""
        print("  - Running homogeneity tests...")
        from scipy.stats import levene, bartlett
        
        results = {}
        continuous_vars = [
            'water_accessibility_score', 'vulnerability_index', 'coping_capabilities_index',
            'wealth_score_continuous', 'infrastructure_asset_score', 'housing_quality_score'
        ]
        grouping_vars = [
            'residence_type', 'wealth_quintile', 'region', 'season',
            'female_headed_household', 'marginalized_caste', 'has_improved_source'
        ]
        
        test_pairs = []
        for cont_var in continuous_vars:
            for group_var in grouping_vars:
                test_pairs.append((cont_var, group_var))

        with ProcessPoolExecutor(max_workers=min(len(test_pairs), self.config.n_cores)) as executor:
            futures = {}
            for cont_var, group_var in test_pairs:
                futures[executor.submit(self._run_single_homogeneity_test, cont_var, group_var)] = (cont_var, group_var)
            
            for future in as_completed(futures):
                cont_var, group_var = futures[future]
                try:
                    result = future.result()
                    results[f'{cont_var}_by_{group_var}'] = result
                except Exception as e:
                    results[f'{cont_var}_by_{group_var}'] = {'error': f'Failed homogeneity test for {cont_var} by {group_var}: {str(e)}'}
                    print(f"  - Failed homogeneity test for {cont_var} by {group_var}: {str(e)}")
        
        return results

    def _run_single_homogeneity_test(self, cont_var: str, group_var: str) -> Dict:
        """Helper to run homogeneity tests for a single variable pair."""
        from scipy.stats import levene, bartlett
        
        if cont_var not in self.df.columns or not self.df[cont_var].notna().any():
            return {'error': f'Continuous variable "{cont_var}" not found or all NA.'}
        if group_var not in self.df.columns or self.df[group_var].nunique(dropna=True) < 2:
            return {'error': f'Grouping variable "{group_var}" not found or has <2 unique values.'}
        
        temp_df = self.df[[cont_var, group_var]].dropna()
        if len(temp_df) < self.config.min_sample * 2:
            return {'error': f'Insufficient data for homogeneity test ({len(temp_df)} rows).'}

        groups_data = []
        group_labels = []
        for category in temp_df[group_var].unique():
            category_data = temp_df[temp_df[group_var] == category][cont_var].dropna()
            if len(category_data) >= self.config.min_sample:
                groups_data.append(category_data)
                group_labels.append(category)
        
        if len(groups_data) < 2:
            return {'error': f'Not enough valid groups ({len(groups_data)}) for homogeneity test.'}

        try:
            levene_stat, levene_p = levene(*groups_data)
            bartlett_stat, bartlett_p = bartlett(*groups_data)
            
            return {
                'levene_statistic': levene_stat,
                'levene_p_value': levene_p,
                'bartlett_statistic': bartlett_stat,
                'bartlett_p_value': bartlett_p,
                'groups_analyzed': group_labels,
                'group_variances': {label: data.var() for label, data in zip(group_labels, groups_data)},
                'homogeneity_levene_p_gt_alpha': levene_p >= self.config.alpha,
                'homogeneity_bartlett_p_gt_alpha': bartlett_p >= self.config.alpha
            }
        except Exception as e:
            return {'error': f'Homogeneity test failed: {str(e)}'}

    def _effect_size_calculations(self) -> Dict:
        """Calculate various effect sizes (Cohen's d, Eta-squared) for key comparisons."""
        print("  - Running effect size calculations...")
        results = {}
        
        target_var = 'water_disrupted'
        continuous_vars_for_d = [
            'water_accessibility_score', 'vulnerability_index', 'coping_capabilities_index',
            'wealth_score_continuous', 'infrastructure_asset_score', 'housing_quality_score',
            'time_to_water_minutes'
        ]
        
        if target_var not in self.df.columns or not self.df[target_var].notna().any():
            return {'error': f'Target variable "{target_var}" not available for effect size calculation.'}

        cohens_d_tasks = []
        for var in continuous_vars_for_d:
            if var in self.df.columns and self.df[var].notna().any():
                cohens_d_tasks.append(var)

        with ProcessPoolExecutor(max_workers=min(len(cohens_d_tasks), self.config.n_cores)) as executor:
            futures = {}
            for var in cohens_d_tasks:
                futures[executor.submit(self._run_single_cohens_d_calculation, var, target_var)] = var
            
            for future in as_completed(futures):
                var_name = futures[future]
                try:
                    results[f'{var_name}_cohens_d_disruption'] = future.result()
                except Exception as e:
                    results[f'{var_name}_cohens_d_disruption'] = {'error': f'Failed Cohen\'s d for {var_name}: {str(e)}'}
                    print(f"  - Failed Cohen\'s d for {var_name}: {str(e)}")
        
        return results

    def _run_single_cohens_d_calculation(self, var: str, target_var: str) -> Dict:
        """Helper to run Cohen's d calculation for a single variable."""
        temp_df = self.df[[var, target_var]].dropna()
        group_disrupted = temp_df[temp_df[target_var] == True][var].dropna()
        group_not_disrupted = temp_df[temp_df[target_var] == False][var].dropna()
        
        if len(group_disrupted) < self.config.min_sample or len(group_not_disrupted) < self.config.min_sample:
            return {'error': f'Insufficient samples for Cohen\'s d ({len(group_not_disrupted)} vs {len(group_disrupted)}).'}

        try:
            n1, n2 = len(group_not_disrupted), len(group_disrupted)
            s1, s2 = group_not_disrupted.std(), group_disrupted.std()
            
            if s1 == 0 and s2 == 0:
                cohens_d = 0.0
            elif s1 == 0:
                 pooled_std = s2
            elif s2 == 0:
                 pooled_std = s1
            else:
                pooled_std = np.sqrt(((n1 - 1) * s1**2 + (n2 - 1) * s2**2) / (n1 + n2 - 2)) if (n1 + n2 - 2) > 0 else 0
            
            cohens_d = (group_disrupted.mean() - group_not_disrupted.mean()) / pooled_std if pooled_std > 0 else 0.0
            
            return {
                'effect_size_type': 'Cohen\'s d',
                'variable': var,
                'group_comparison': 'Disrupted vs Not Disrupted',
                'cohens_d': cohens_d,
                'interpretation': self._interpret_cohens_d(cohens_d),
                'mean_disrupted': group_disrupted.mean(),
                'mean_not_disrupted': group_not_disrupted.mean(),
                'n_disrupted': n2,
                'n_not_disrupted': n1
            }
        except Exception as e:
            return {'error': f'Cohen\'s d calculation failed: {str(e)}'}

    def _interpret_cohens_d(self, d: float) -> str:
        """Interpret Cohen's d magnitude."""
        abs_d = abs(d)
        if abs_d < 0.2: return 'negligible'
        elif abs_d < 0.5: return 'small'
        elif abs_d < 0.8: return 'medium'
        else: return 'large'

    def _power_analysis(self) -> Dict:
        """Basic power analysis for key comparisons (e.g., difference in means)."""
        print("  - Running power analysis...")
        results = {}
        
        target_var = 'water_disrupted'
        access_score_var = 'water_accessibility_score'
        
        if target_var in self.df.columns and access_score_var in self.df.columns:
            temp_df = self.df[[target_var, access_score_var]].dropna()
            group_disrupted = temp_df[temp_df[target_var] == True][access_score_var].dropna()
            group_not_disrupted = temp_df[temp_df[target_var] == False][access_score_var].dropna()
            
            if len(group_disrupted) >= self.config.min_sample and len(group_not_disrupted) >= self.config.min_sample:
                try:
                    from statsmodels.stats.power import TTestIndPower
                    
                    n1, n2 = len(group_not_disrupted), len(group_disrupted)
                    m1, m2 = group_not_disrupted.mean(), group_disrupted.mean()
                    s1, s2 = group_not_disrupted.std(), group_disrupted.std()
                    
                    if s1 == 0 and s2 == 0:
                        observed_d = 0.0
                    elif s1 == 0:
                        pooled_std = s2
                    elif s2 == 0:
                        pooled_std = s1
                    else:
                        pooled_std = np.sqrt(((n1 - 1) * s1**2 + (n2 - 1) * s2**2) / (n1 + n2 - 2)) if (n1 + n2 - 2) > 0 else 0
                    
                    observed_d = (m2 - m1) / pooled_std if pooled_std > 0 else 0.0
                    
                    if np.isfinite(observed_d) and observed_d != 0:
                        analysis = TTestIndPower()
                        power = analysis.power(
                            effect_size=observed_d,
                            nobs1=n1,
                            ratio=n2/n1,
                            alpha=self.config.alpha,
                            alternative='two-sided'
                        )
                        results['power_analysis_disruption_access_score'] = {
                            'variable_compared': access_score_var,
                            'target_variable': target_var,
                            'observed_cohens_d': observed_d,
                            'sample_size_group1': n1,
                            'sample_size_group2': n2,
                            'alpha_level': self.config.alpha,
                            'estimated_power': power,
                            'interpretation': 'High power indicates good chance of detecting observed effect. Low power means large effect might be missed.'
                        }
                    else:
                        results['power_analysis_disruption_access_score'] = {'note': 'Cannot perform power analysis: observed effect size is zero or infinite.'}
                except Exception as e:
                    results['power_analysis_disruption_access_score'] = {'error': f'Power analysis failed: {str(e)}'}
        
        results['overall_sample_size_summary'] = {
            'total_households': len(self.df),
            'total_weighted_population': self.df['original_weight'].sum() if 'original_weight' in self.df.columns else np.nan,
            'disruption_rate_unweighted_known_status': self.df['water_disrupted'].mean() * 100 if 'water_disrupted' in self.df.columns else np.nan,
            'disruption_rate_weighted_known_status': safe_weighted_average(self.df.loc[self.df['water_disrupted'].notna()], 'water_disrupted', 'original_weight') * 100 if 'water_disrupted' in self.df.columns and 'original_weight' in self.df.columns else np.nan
        }
        
        return results

    def _multiple_comparisons_correction(self) -> Dict:
        """Apply multiple comparisons corrections (Bonferroni, Benjamini-Hochberg) to collected p-values."""
        print("  - Running multiple comparisons correction...")
        results = {}
        
        if self.bivariate_results is None:
            return {'error': 'Bivariate results not provided for multiple comparisons correction.'}
        p_values_chi2 = []
        test_names_chi2 = []
        if 'chi_square_tests' in self.bivariate_results:
            for var, test_res in self.bivariate_results['chi_square_tests'].items():
                if isinstance(test_res, dict) and 'p_value' in test_res and not np.isnan(test_res['p_value']):
                    p_values_chi2.append(test_res['p_value'])
                    test_names_chi2.append(f'Chi2_{var}')
        
        p_values_ttest = []
        test_names_ttest = []
        if 't_tests' in self.bivariate_results:
            for var, test_res in self.bivariate_results['t_tests'].items():
                if isinstance(test_res, dict) and 'p_value' in test_res and not np.isnan(test_res['p_value']):
                    p_values_ttest.append(test_res['p_value'])
                    test_names_ttest.append(f'Ttest_{var}')
        p_values_anova = []
        test_names_anova = []
        if 'anova_tests' in self.bivariate_results:
            for var, test_res in self.bivariate_results['anova_tests'].items():
                if isinstance(test_res, dict) and 'f_p_value' in test_res and not np.isnan(test_res['f_p_value']):
                    p_values_anova.append(test_res['f_p_value'])
                    test_names_anova.append(f'ANOVA_{var}')
        
        all_p_values = p_values_chi2 + p_values_ttest + p_values_anova
        all_test_names = test_names_chi2 + test_names_ttest + test_names_anova
        
        if len(all_p_values) > 1:
            try:
                reject_bonferroni, p_corrected_bonferroni, _, _ = multipletests(all_p_values, alpha=self.config.alpha, method='bonferroni')
                reject_bh, p_corrected_bh, _, _ = multipletests(all_p_values, alpha=self.config.alpha, method='fdr_bh')
                
                correction_summary = pd.DataFrame({
                    'Test_Name': all_test_names,
                    'Original_P_Value': all_p_values,
                    'Bonferroni_Corrected_P': p_corrected_bonferroni,
                    'Bonferroni_Rejected': reject_bonferroni,
                    'Benjamini_Hochberg_Corrected_P': p_corrected_bh,
                    'Benjamini_Hochberg_Rejected': reject_bh
                })
                
                results['multiple_comparisons_summary'] = correction_summary
                results['n_tests_corrected'] = len(all_p_values)
                results['alpha_level'] = self.config.alpha
            except Exception as e:
                results['error'] = f'Multiple comparisons correction failed: {str(e)}'
        else:
            results['multiple_comparisons_summary'] = {'note': 'Not enough p-values (needs >1) for multiple comparisons correction.'}
        
        return results

    def _create_executive_summary(self, writer):
        """Creates the executive summary sheet with key metrics."""
        
        summary_data = {
            'Metric': [],
            'Value': []
        }
        
        total_hh = len(self.df)
        weighted_pop = self.df['original_weight'].sum() if 'original_weight' in self.df.columns else np.nan
        
        unweighted_disruption_rate = self.df['water_disrupted'].mean() * 100 if 'water_disrupted' in self.df.columns else np.nan
        
        weighted_disruption_rate_known_status = safe_weighted_average(self.df.loc[self.df['water_disrupted'].notna()], 'water_disrupted', 'original_weight') * 100 if 'water_disrupted' in self.df.columns and 'original_weight' in self.df.columns else np.nan
        
        weighted_true_sum = self.df.loc[self.df['water_disrupted'] == True, 'original_weight'].sum()
        total_weight_sum_for_disruption = self.df['original_weight'].sum()
        
        if total_weight_sum_for_disruption > 0:
            weighted_disruption_rate_total_sample = (weighted_true_sum / total_weight_sum_for_disruption) * 100
        else:
            weighted_disruption_rate_total_sample = np.nan

        disrupted_hh_count = self.df['water_disrupted'].sum() if 'water_disrupted' in self.df.columns else np.nan
        estimated_affected_pop = (self.df['water_disrupted'] * self.df['original_weight']).sum() if 'water_disrupted' in self.df.columns and 'original_weight' in self.df.columns else np.nan
        
        summary_data['Metric'].extend([
            'Total Households (Sample)', 'Estimated Population (Weighted)',
            'Water Disruption Rate - Unweighted (Known Status) (%)',
            'Water Disruption Rate - Unweighted (Total Sample) (%)',
            'Water Disruption Rate - Weighted (Known Status) (%)',
            'Water Disruption Rate - Weighted (Total Sample) (%)',
            'Households with Water Disruption (Count)',
            'Estimated People Affected (Weighted Count)'
        ])
        summary_data['Value'].extend([
            f"{total_hh:,}",
            f"{weighted_pop:,.0f}",
            f"{unweighted_disruption_rate:.2f}",
            f"{self.df['water_disrupted'].mean()*100:.2f}",
            f"{weighted_disruption_rate_known_status:.2f}",
            f"{weighted_disruption_rate_total_sample:.2f}",
            f"{int(disrupted_hh_count):,}",
            f"{estimated_affected_pop:,.0f}"
        ])
        
        if 'residence_type' in self.df.columns:
            summary_data['Metric'].append('')
            summary_data['Value'].append('')
            summary_data['Metric'].append('Urban/Rural Breakdown:')
            summary_data['Value'].append('')
            for res_type in self.df['residence_type'].unique():
                if pd.notna(res_type):
                    subset = self.df[self.df['residence_type'] == res_type]
                    uw_rate = subset['water_disrupted'].mean() * 100 if 'water_disrupted' in subset.columns else np.nan
                    w_rate = safe_weighted_average(subset, 'water_disrupted', 'original_weight') * 100 if 'water_disrupted' in subset.columns and 'original_weight' in subset.columns else np.nan
                    
                    summary_data['Metric'].append(f"{res_type} Households (Count)")
                    summary_data['Value'].append(f"{len(subset):,}")
                    summary_data['Metric'].append(f"{res_type} Disruption Rate - Unweighted (%)")
                    summary_data['Value'].append(f"{uw_rate:.2f}")
                    summary_data['Metric'].append(f"{res_type} Disruption Rate - Weighted (%)")
                    summary_data['Value'].append(f"{w_rate:.2f}")
        
        if 'region' in self.df.columns:
            summary_data['Metric'].append('')
            summary_data['Value'].append('')
            summary_data['Metric'].append('Regional Disruption Rates:')
            summary_data['Value'].append('')
            
            region_disruption = self.df.groupby('region').apply(
                lambda x: pd.Series({
                    'Unweighted_Rate': x['water_disrupted'].mean() * 100,
                    'Weighted_Rate': safe_weighted_average(x, 'water_disrupted', 'original_weight') * 100,
                    'Count': len(x)
                })
            ).reset_index()
            
            for _, row in region_disruption.iterrows():
                summary_data['Metric'].append(f"{row['region']} (N={int(row['Count']):,}) - Unweighted (%)")
                summary_data['Value'].append(f"{row['Unweighted_Rate']:.2f}")
                summary_data['Metric'].append(f"{row['region']} (N={int(row['Count']):,}) - Weighted (%)")
                summary_data['Value'].append(f"{row['Weighted_Rate']:.2f}")

        summary_data['Metric'].extend(['', 'Study Parameters:'])
        summary_data['Value'].extend(['', ''])
        summary_data['Metric'].append('States Covered')
        summary_data['Value'].append(self.df['state_name'].nunique() if 'state_name' in self.df.columns else 'N/A')
        summary_data['Metric'].append('Clusters Covered')
        summary_data['Value'].append(self.df['hv001'].nunique() if 'hv001' in self.df.columns else 'N/A')
        summary_data['Metric'].append('Interview Years')
        summary_data['Value'].append(f"{self.df['interview_year'].min():.0f}-{self.df['interview_year'].max():.0f}" if 'interview_year' in self.df.columns and self.df['interview_year'].notna().any() else 'N/A')
        summary_data['Metric'].append('Seasons Covered')
        summary_data['Value'].append(', '.join(self.df['season'].dropna().unique()) if 'season' in self.df.columns else 'N/A')
        pd.DataFrame(summary_data).to_excel(writer, sheet_name='Executive_Summary', index=False)

    def _create_sample_sizes_sheet(self, writer):
        """Creates a sheet detailing sample sizes for each variable and category."""
        sample_size_records = []
        for var_name, data in self.processor_sample_sizes.items():
            if 'error' in data:
                sample_size_records.append({
                    'Variable': var_name, 'Category': 'Error', 'Unweighted_Count': data['error'],
                    'Unweighted_Percent': '', 'Weighted_Count': '', 'Weighted_Percent': ''
                })
                continue
            
            summary = data['summary']
            sample_size_records.append({
                'Variable': var_name.upper(), 'Category': 'TOTAL VALID',
                'Unweighted_Count': summary['valid_unweighted_n'],
                'Unweighted_Percent': f"{summary['valid_unweighted_percent']:.2f}%",
                'Weighted_Count': f"{summary['valid_weighted_n']:.0f}",
                'Weighted_Percent': f"{summary['valid_weighted_percent']:.2f}%"
            })
            sample_size_records.append({
                'Variable': var_name.upper(), 'Category': 'MISSING',
                'Unweighted_Count': summary['missing_unweighted_n'],
                'Unweighted_Percent': f"{summary['missing_unweighted_percent']:.2f}%",
                'Weighted_Count': f"{summary['missing_weighted_n']:.0f}",
                'Weighted_Percent': f"{summary['missing_weighted_percent']:.2f}%"
            })
            for cat_label, cat_data in data['categories'].items():
                sample_size_records.append({
                    'Variable': var_name, 'Category': cat_label,
                    'Unweighted_Count': cat_data['unweighted_count'],
                    'Unweighted_Percent': f"{cat_data['unweighted_percent']:.2f}%",
                    'Weighted_Count': f"{cat_data['weighted_count']:.0f}",
                    'Weighted_Percent': f"{cat_data['weighted_percent']:.2f}%"
                })
            sample_size_records.append({})
        pd.DataFrame(sample_size_records).to_excel(writer, sheet_name='Sample_Sizes_by_Variable', index=False)

    def _create_descriptive_statistics(self, writer):
        """Creates a sheet with descriptive statistics of water disruption by various categorical variables."""
        desc_data_records = []
        analysis_vars = [
            'residence_type', 'wealth_quintile', 'region', 'season', 'water_source_category',
            'caste_category', 'religion', 'hh_head_education_level', 'toilet_type',
            'time_to_water_category', 'water_fetcher', 'water_source_location',
            'place_type', 'has_improved_source', 'women_fetch_water', 'children_fetch_water',
            'has_electricity', 'has_improved_sanitation', 'practices_open_defecation',
            'has_bank_account', 'has_separate_kitchen', 'has_vehicle', 'has_finished_floor',
            'has_finished_wall', 'has_finished_roof', 'female_headed_household', 'marginalized_caste'
        ]
        
        target_var = 'water_disrupted'
        for var in analysis_vars:
            if var not in self.df.columns or not self.df[target_var].notna().any():
                continue
            desc_data_records.append({'Variable': var.upper(), 'Category': 'RAW DATA FOR CALCULATION'})
            
            temp_df = self.df[[var, target_var, 'original_weight']].copy()
            
            group_stats = temp_df.groupby(var).apply(
                lambda x: pd.Series({
                    'N_Households': len(x),
                    'Unweighted_Disrupted_Count': x[target_var].sum(),
                    'Unweighted_Disruption_Rate': x[target_var].mean() * 100,
                    'Weighted_Disrupted_Count': (x[target_var] * x['original_weight']).sum(),
                    'Weighted_Disruption_Rate': safe_weighted_average(x, target_var, 'original_weight') * 100
                })
            ).reset_index()
            
            # Identify numeric columns for fillna(0), avoiding categorical columns like 'Category'
            numeric_cols = group_stats.select_dtypes(include=np.number).columns
            group_stats[numeric_cols] = group_stats[numeric_cols].fillna(0)
            
            group_stats.rename(columns={var: 'Category'}, inplace=True)
            group_stats.insert(0, 'Variable', var)
            
            desc_data_records.extend(group_stats.to_dict('records'))
            desc_data_records.append({})
        pd.DataFrame(desc_data_records).to_excel(writer, sheet_name='Descriptive_Statistics', index=False)

    def _create_statewise_analysis(self, writer):
        """Creates a sheet with detailed state-wise analysis of water disruption."""
        if 'state_name' not in self.df.columns:
            pd.DataFrame({'Message': ['State data not available for analysis.']}).to_excel(
                writer, sheet_name='State_Analysis', index=False)
            return
        state_records = []
        target_var = 'water_disrupted'
        
        state_stats = self.df.groupby('state_name').apply(
            lambda x: pd.Series({
                'Total_Households': len(x),
                'Weighted_Population': x['original_weight'].sum(),
                'Disrupted_Households_Unweighted': x[target_var].sum(),
                'Disruption_Rate_Unweighted': x[target_var].mean() * 100,
                'Disruption_Rate_Weighted': safe_weighted_average(x, target_var, 'original_weight') * 100,
                'Urban_Households': len(x[x['residence_type'] == 'Urban']) if 'residence_type' in x.columns else 0,
                'Rural_Households': len(x[x['residence_type'] == 'Rural']) if 'residence_type' in x.columns else 0,
                'Urban_Disruption_Rate_Unweighted': x[x['residence_type'] == 'Urban'][target_var].mean() * 100 if 'residence_type' in x.columns and not x[x['residence_type'] == 'Urban'].empty else np.nan,
                'Rural_Disruption_Rate_Unweighted': x[x['residence_type'] == 'Rural'][target_var].mean() * 100 if 'residence_type' in x.columns and not x[x['residence_type'] == 'Rural'].empty else np.nan
            })
        ).reset_index()
        
        state_stats = state_stats.sort_values('Disruption_Rate_Weighted', ascending=False)
        
        state_records.extend(state_stats.to_dict('records'))
        pd.DataFrame(state_records).to_excel(writer, sheet_name='State_Analysis', index=False)

            # Write data
            for r_idx, row_data in enumerate(df_to_write.values):
                for c_idx, cell_value in enumerate(row_data):
                    ws.cell(row=start_row + r_idx + 2, column=start_col + c_idx + 1, value=cell_value)
            
            return start_row + len(df_to_write) + 3

        from openpyxl.styles import Font

        if self.bivariate_results.get('chi_square_tests'):
            worksheet.cell(row=row_offset + 1, column=1, value='Chi-Square Tests (Categorical vs Water Disruption)').font = Font(bold=True, size=14)

                        'Significant': res['significant']
                    })
                else:
                    chi2_data.append({'Variable': var, 'Error': res['error']})
            if chi2_data:
                chi2_df = pd.DataFrame(chi2_data)
                row_offset = write_df_to_ws(worksheet, chi2_df, row_offset, 0)
                row_offset += 1

        if self.bivariate_results.get('t_tests'):
            worksheet.cell(row=row_offset + 1, column=1, value='T-Tests (Continuous vs Water Disruption Groups)').font = Font(bold=True, size=14)
            row_offset += 2
            
            ttest_data = []
            for var, res in self.bivariate_results['t_tests'].items():
                if 'error' not in res:
                    ttest_data.append({
                        'Variable': var,
                        'Group0_Mean': f"{res['group0_mean']:.3f}",
                        'Group1_Mean': f"{res['group1_mean']:.3f}",
                        'T_Statistic': f"{res['t_statistic']:.3f}",

                row_offset = write_df_to_ws(worksheet, ttest_df, row_offset, 0)
                row_offset += 1

        if self.bivariate_results.get('anova_tests'):
            worksheet.cell(row=row_offset + 1, column=1, value='ANOVA Tests (Continuous by Categorical Group)').font = Font(bold=True, size=14)
            row_offset += 2
            
            anova_data = []
            for var_combo, res in self.bivariate_results['anova_tests'].items():
                if 'error' not in res:
                    anova_data.append({
                        'Variable_Combination': var_combo,
                        'F_Statistic': f"{res['f_statistic']:.3f}",
                        'F_P_Value': f"{res['f_p_value']:.4f}",

                        'Kruskal_H': f"{res['kruskal_h']:.3f}",
                        'Kruskal_P': f"{res['kruskal_p']:.4f}",
                        'Significant_ANOVA': res['significant_anova'],
                        'Significant_Kruskal': res['significant_kruskal']
                    })
                else:
                    anova_data.append({'Variable_Combination': var_combo, 'Error': res['error']})

                row_offset += 1

        if self.bivariate_results.get('correlation_analysis'):
            corr_res = self.bivariate_results['correlation_analysis']
            for corr_type in ['pearson', 'spearman', 'kendall']:
                if f'{corr_type}_correlation_matrix' in corr_res:
                    corr_matrix = corr_res[f'{corr_type}_correlation_matrix']
                    p_matrix = corr_res[f'{corr_type}_p_value_matrix']
                    
                    combined_corr_df = corr_matrix.copy().astype(str)
                    for col in combined_corr_df.columns:
                        for idx in combined_corr_df.index:
                            r_val = corr_matrix.loc[idx, col]
                            p_val = p_matrix.loc[idx, col]
                            if pd.isna(r_val):
                                combined_corr_df.loc[idx, col] = 'NA'
                            elif idx == col:
                                combined_corr_df.loc[idx, col] = '1.000'

                    row_offset = write_df_to_ws(worksheet, row_pct_df_reset, row_offset, 0)

    def _create_raw_data_summaries(self, writer):
        """Creates sheets with summary statistics for raw input variables."""
        worksheet = writer.book.create_sheet('Raw_Data_Summaries')
        row_offset = 0

        def write_df_to_ws(ws, df_to_write, start_row, start_col):
            # Write header
            for col_idx, col_name in enumerate(df_to_write.columns):
                ws.cell(row=start_row + 1, column=start_col + col_idx + 1, value=col_name).font = Font(bold=True)
            
            # Write data
            for r_idx, row_data in enumerate(df_to_write.values):
                for c_idx, cell_value in enumerate(row_data):
                    ws.cell(row=start_row + r_idx + 2, column=start_col + c_idx + 1, value=cell_value)
            
            return start_row + len(df_to_write) + 3

        from openpyxl.styles import Font

        raw_vars_to_summarize = [
            'hv005', 'hv008', 'hv025', 'hv024', 'hv201', 'hv204', 'hv270', 'hv009', 'hv014',
            'hv206', 'hv205', 'hv243a', 'sh37b', 'hv201a'
        ]
        
        for var in raw_vars_to_summarize:
            if var not in self.df.columns:
                worksheet.cell(row=row_offset + 1, column=1, value=f"Variable: {var} (Not Found)").font = Font(bold=True, italic=True)
                row_offset += 2
                continue
            
            worksheet.cell(row=row_offset + 1, column=1, value=f"Summary for: {var}").font = Font(bold=True, size=14)
            row_offset += 2

            series = self.df[var].copy()
            
            if pd.api.types.is_numeric_dtype(series):
                worksheet.cell(row=row_offset + 1, column=1, value='Numeric Summary').font = Font(bold=True)
                row_offset += 1
                summary = series.describe().reset_index()
                summary.columns = ['Statistic', 'Value']
                row_offset = write_df_to_ws(worksheet, summary, row_offset, 0)
                
                if series.nunique() < 50:
                    worksheet.cell(row=row_offset + 1, column=1, value='Value Counts (Unweighted)').font = Font(bold=True)
                    row_offset += 1
                    counts = series.value_counts(dropna=False).reset_index()
                    counts.columns = ['Value', 'Count']
                    counts['Percentage'] = counts['Count'] / counts['Count'].sum() * 100
                    row_offset = write_df_to_ws(worksheet, counts, row_offset, 0)
            
            else:
                worksheet.cell(row=row_offset + 1, column=1, value='Value Counts (Unweighted)').font = Font(bold=True)
                row_offset += 1
                counts = series.value_counts(dropna=False).reset_index()
                counts.columns = ['Value', 'Count']
                counts['Percentage'] = counts['Count'] / counts['Count'].sum() * 100
                row_offset = write_df_to_ws(worksheet, counts, row_offset, 0)
            row_offset += 1

    def _create_correlation_heatmap(self) -> str:
        """Generates a heatmap of Pearson correlations for key numerical variables."""
        if not self.bivariate_results.get('correlation_analysis') or 'pearson_correlation_matrix' not in self.bivariate_results['correlation_analysis']:
            print("Pearson correlation matrix not available for heatmap.")
            return "correlation_heatmap_not_available.png"
        
        corr_matrix = self.bivariate_results['correlation_analysis']['pearson_correlation_matrix']
        
        display_cols = [
            'water_disrupted', 'water_disruptive_score', 'water_accessibility_score',
            'vulnerability_index', 'coping_capabilities_index',
            'wealth_score_continuous', 'infrastructure_asset_score', 'housing_quality_score',
            'time_to_water_minutes', 'hh_size', 'children_under5_count'
        ]
        
        display_corr_matrix = corr_matrix.loc[
            [c for c in display_cols if c in corr_matrix.columns],
            [c for c in display_cols if c in corr_matrix.columns]
        ]
        
        filename = os.path.join(self.output_dir, 'correlation_heatmap.png')
        
        if not display_corr_matrix.empty:
            fig, ax = plt.subplots(figsize=(10, 8)) # Define fig here
            sns.heatmap(display_corr_matrix, annot=True, cmap='coolwarm', fmt=".2f", linewidths=.5, ax=ax)
            plt.title('Pearson Correlation Heatmap of Key Variables')
            plt.xticks(rotation=45, ha='right')
            plt.yticks(rotation=0)
            plt.tight_layout()
            plt.savefig(filename, dpi=300)
            plt.close(fig) # Close fig here
        else:
            print("Filtered correlation matrix is empty. Not generating heatmap.")
            # No figure created, so no need to close
        
        return filename

    def _create_cluster_profile_plot(self) -> str:
        """Generates a radar plot or bar plot to visualize cluster profiles."""
        if not self.multivariate_results.get('cluster_analysis') or 'final_clusters' not in self.multivariate_results['cluster_analysis']:
            print("Cluster analysis results not available for plotting.")
            return "cluster_profile_plot_not_available.png"
        
        cluster_res = self.multivariate_results['cluster_analysis']
        cluster_profiles_df = cluster_res['final_clusters']['cluster_profiles_mean']
        
        features_to_plot = cluster_profiles_df.columns.tolist()

        ax.tick_params(axis='x', rotation=90)

        plt.tight_layout()
        filename = os.path.join(self.output_dir, 'cluster_profile_plot.png')
        plt.savefig(filename, dpi=300)
        plt.close(fig)
        return filename

def main():
    """Main execution function for the comprehensive water disruption analysis."""
    print("=" * 60)
    print("NFHS 2019-21 Comprehensive Water Disruption Analysis")
    print("Enhanced with Bivariate and Multivariate Statistical Analyses")
    print("=" * 60)
    
    config = Config()

    file_path_to_load = DATA_FILE_PATH
    if not os.path.exists(file_path_to_load):
        print(f"Data file not found at configured path: {file_path_to_load}")
        file_path_to_load = input("Please enter the full path to the NFHS DTA file: ").strip().strip('"')
        if not os.path.exists(file_path_to_load):
            print(f"Error: File not found at {file_path_to_load}. Exiting.")
            return

    loader = EnhancedDataLoader(config)
    df = loader.load(file_path_to_load)

    if df.empty:
        print("No data loaded. Exiting analysis.")
        return

    processor = EnhancedWaterDisruptionProcessor(df)
    df_processed = processor.prepare()

    if df_processed.empty:
        print("Data processing resulted in an empty DataFrame. Exiting analysis.")
        return

    print("\n" + "="*50)
    print("RUNNING COMPREHENSIVE STATISTICAL ANALYSES (PARALLELIZED)")
    print("="*50)

    temp_results_dir = "temp_analysis_results"
    os.makedirs(temp_results_dir, exist_ok=True)
    
    analysis_results = {}

    tasks_config = {
        'bivariate': {
            'class': BivariateAnalyzer,
            'method': 'run_comprehensive_bivariate_analysis',
            'temp_file': os.path.join(temp_results_dir, 'bivariate_results.pkl')
        },
        'multivariate': {
            'class': MultivariateAnalyzer,
            'method': 'run_comprehensive_multivariate_analysis',
            'temp_file': os.path.join(temp_results_dir, 'multivariate_results.pkl')
        },
        'advanced': {
            'class': AdvancedStatisticalTests,
            'method': 'run_advanced_tests',
            'temp_file': os.path.join(temp_results_dir, 'advanced_results.pkl')
        }
    }

    with ProcessPoolExecutor(max_workers=config.n_cores) as executor:
        futures = {}
        for name, task_info in tasks_config.items():
            if os.path.exists(task_info['temp_file']):
                print(f"Loading '{name}' results from {task_info['temp_file']}")
                try:
                    with open(task_info['temp_file'], 'rb') as f:
                        analysis_results[name] = pickle.load(f)
                    print(f"Successfully loaded '{name}' results.")
                    continue
                except Exception as e:
                    print(f"Error loading '{name}' results from temp file: {e}. Recalculating.")
            
            instance = task_info['class'](df_processed, config)
            futures[name] = executor.submit(getattr(instance, task_info['method']))

        for name, future in futures.items():
            try:
                result = future.result()
                analysis_results[name] = result
                print(f"Completed '{name}' analysis.")
                
                with open(tasks_config[name]['temp_file'], 'wb') as f:
                    pickle.dump(result, f)
                print(f"Saved '{name}' results to {tasks_config[name]['temp_file']}")

            except Exception as e:
                print(f"Error during '{name}' analysis: {e}")
                analysis_results[name] = {'error': f"Analysis failed: {e}"}
    
    if 'multivariate' in analysis_results and 'cluster_analysis' in analysis_results['multivariate']:
        cluster_res = analysis_results['multivariate']['cluster_analysis']
        if 'error' not in cluster_res and 'final_clusters' in cluster_res:
            final_clusters_data = cluster_res['final_clusters']
            cluster_labels_index = final_clusters_data.get('cluster_labels_index')
            cluster_labels_values = final_clusters_data.get('cluster_labels_values')

            if cluster_labels_index and cluster_labels_values:
                print("Applying coping_capability_cluster labels to main DataFrame...")
                df_processed['coping_capability_cluster'] = pd.Series(index=df_processed.index, dtype=int)
                df_processed.loc[cluster_labels_index, 'coping_capability_cluster'] = cluster_labels_values
                print("Coping capability clusters applied.")
            else:
                print("Could not retrieve cluster labels for application.")
        else:
            print("Skipping coping_capability_cluster application due to cluster analysis error or missing results.")

    if 'bivariate' in analysis_results and 'advanced' in analysis_results:
        temp_advanced_instance = AdvancedStatisticalTests(df_processed, config)
        temp_advanced_instance.bivariate_results = analysis_results['bivariate'] 
        
        if 'error' not in analysis_results['advanced']:
             print("Re-running Multiple Comparisons Correction to ensure context with bivariate results.")
             mcc_results = temp_advanced_instance._multiple_comparisons_correction()
             analysis_results['advanced']['multiple_comparisons_correction'] = mcc_results
             
             with open(tasks_config['advanced']['temp_file'], 'wb') as f:
                 pickle.dump(analysis_results['advanced'], f)
             print(f"Updated '{'advanced'}' results in {tasks_config['advanced']['temp_file']} with MCC.")
        else:
            print("Skipping MCC as advanced analysis had a prior error.")

    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    os.makedirs(VISUALIZATION_OUTPUT_DIR, exist_ok=True)

    viz_generator = EnhancedVisualizationGenerator(
        df=df_processed,
        processor_sample_sizes=processor.sample_sizes,
        bivariate_results=analysis_results.get('bivariate', {}),
        multivariate_results=analysis_results.get('multivariate', {}),
        advanced_results=analysis_results.get('advanced', {}),
        output_dir=VISUALIZATION_OUTPUT_DIR
    )
    viz_files = viz_generator.generate_all_visualizations()

    report_generator = ComprehensiveReportGenerator(
        df=df_processed,
        processor_sample_sizes=processor.sample_sizes,
        bivariate_results=analysis_results.get('bivariate', {}),
        multivariate_results=analysis_results.get('multivariate', {}),
        advanced_results=analysis_results.get('advanced', {}),
        config=config
    )
    report_filename = report_generator.generate_comprehensive_report(timestamp)

    print("\n" + "="*60)
    print("COMPREHENSIVE ANALYSIS COMPLETE!")
    print("="*60)
    
    print(f"\nKEY FINDINGS SUMMARY:")
    
    # Initialize variables to 'N/A' or appropriate default values
    overall_unweighted_rate_known_status = np.nan
    overall_unweighted_rate_total_sample = np.nan
    overall_weighted_rate_known_status = np.nan
    overall_weighted_rate_total_sample = np.nan # Initialize here
    urban_weighted_rate = np.nan
    rural_weighted_rate = np.nan

    if 'water_disrupted' in df_processed.columns and df_processed['water_disrupted'].notna().any():
        overall_unweighted_rate_known_status = df_processed['water_disrupted'].mean() * 100
        count_disrupted_true = df_processed['water_disrupted'].sum() # Sum of 1s
        overall_unweighted_rate_total_sample = (count_disrupted_true / len(df_processed)) * 100

        valid_disruption = df_processed['water_disrupted'].notna() & df_processed['original_weight'].notna()
        if valid_disruption.any() and df_processed.loc[valid_disruption, 'original_weight'].sum() > 0:
            overall_weighted_rate_known_status = np.average(df_processed.loc[valid_disruption, 'water_disrupted'], weights=df_processed.loc[valid_disruption, 'original_weight']) * 100
        else:
            overall_weighted_rate_known_status = np.nan
